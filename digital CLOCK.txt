function clock() {

var today = new Date();
var date = today.getDate();

var month = today.getMonth();
var allmonths = ["Jan", "Feb","Mar","April","May","June","July","Aug","Sept","Oct", "Nov","Dec"];
month = allmonths[month];

var year = today.getFullYear();

var day = today.getDay();
var alldays = ["Sun","Mon","Tues","Wed","Thurs","Fri","Sat"];
day = alldays[day];

var hour = today.getHours();
var min = today.getMinutes();
var sec = today.getSeconds();
min = checktime(min);
sec = checktime(sec);

document.getElementById("date").innerHTML= day + "," + date + "/" + month + "/" + year;
document.getElementById("time").innerHTML = hour + ":" + min + ":" + sec;
var t = setTimeout(clock,1000);
}

function checktime(i) {
if (i < 10) 
{ i = "0" + i};  // add zero infront of numbers that is less than 10
return i;
}
